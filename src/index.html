JavaScript API
// Create a lightbox instance
const lightbox = new Lightbox('#lightbox-1');

// Open the lightbox
lightbox.open();

// Close the lightbox
lightbox.close();

// Toggle the lightbox
lightbox.toggle();

// Listen to events
element.addEventListener('lightbox:open', () => console.log('Opened'));
element.addEventListener('lightbox:close', () => console.log('Closed'));
